#### 호이스팅 : 자바스크립트는 변수나 함수의 선언부분을 그들이 속한 스코프의 최상단으로 끌어올려지는 특성을 말한다.
```js
// 변수의 호이스팅은 '선언 부분'만 최상단으로 이동된다. 할당은 제 위치에서 이루어진다.
console.log(var1); // 출력 : undefined
var var1 = 5;
console.log(var1); // 출력 : 5

// 함수의 호이스팅 
// (1) function 키워드 : 선언과 동시에 호이스팅되며 초기화가 진행됨.
myFunc(); // 출력 : 'Hello, World' , 호출이 함수 선언전에 이루어졌지만 호이스팅 덕분에 출력이 된다.
function myFunc() {
  	console.log('Hello, World');
}
// (2) arrow function : 선언과 동시에 호이스팅 되지만 초기화는 실제 코드의 위치에서 진행되기 때문에, arrow function의 변수 명은 호이스팅 되지만, 본문은 호이스팅 되지 않는다. 
myFunc(); // 출력 : TypeError: myFunc is not a function.
var myFunc = () => { 
  	console.log('Hello, World');
}
```

#### 전역변수와 변수의 참조
```js
// 변수의 참조 : 변수를 참조한다는 것은 변수가 가르키는 메모리 위치에 저장된 값을 읽거나 쓴다는 것을 의미
// 변수는 참조가능하다. (바깥에 있는 변수는 안쪽으로 자유롭게 사용가능.)
var 나이 = 20;

function 함수() {
  	console.log(나이)
}
함수();
// 함수(){} 안쪽에 나이라는 변수가 없지만, 그 바깥에 있는 변수를 참조한다. 즉 바깥쪽에 선언된 변수를 함수 내부에서도 사용할 수 있다. 

// 전역변수
// 바닐라에서 전역변수 선언하는 방법
<script>
  	var 나이 = 20; // 전역변수, 여기서 선언된 전역변수는 window 객체에 자동으로 저장이 된다.

	function 함수() {
      	console.log(나이)
	}
</script>

```

#### 클로저 : 클로저는 선언된 '함수와' 그 함수가 접근할 수 있는 변수들
```js
// 클로저 사전적인 정의 : 클로저는 함수와 그 함수가 선언된 어휘적 환경의 조합. 이는 함수가 선언된 시점의 스코프 체인을 기억하고, 이 스코프 체인 외부에서 호출되더라도 이 스코프 체인에 접근할 수 있게 해준다.

// 클로저의 기본 개념 : 함수가 어떠한 외부 변수에 접근할 수 있고, 그 함수가 어디서는 호출되든 해당 변수에 접근할 수 있개 하는 것.

function createCounter() {
  	let count = 0;
  	return function() { // 이 부분이 클로져
      	count += 1;
      	console.log(count);
	}
}
// createCounter() 함수는 내부에 count변수를 가지고 있고, 이 변수를 참조하는 또 다른 함수를 리턴해 줍니다. 

let counter = createCounter();
counter(); // 출력 : 1
counter(); // 출력 : 2
// counter 함수를 실행하면 createCounter 내부 변수인 count의 값을 변경하고 출력합니다. 여기서 주목할 점은 createCounter 호출이 끝나고 스코프가 종료된 후에도, counter 함수는 count변수를 계속 참조하고 있다는 점이다. 

// 다음과 같이 함수가 자신ㄴ이 선언된 환경을 '기억'하고, 그 환경에 있는 변수들을 참조할 수 있는 이 특성이 클로저입니다. 이 특성 때문에 함수 범위 밖에서도 변수를 안전하게 참조하고 사용할 수 있습니다.
```
