#### 자바스크립트에서 문자를 다룰 때 어려웠던 점을 보안하기 위해 나온 문법들

#### Backtick

```js
// backtick ``
// backtick 장점(1) : 문자 중간 엔터키 입력 가능.
var letter = `hello
world`;

// backtick 장점(2) : 문자 중간 변수 입력 가능.
var name = '손흥민';
var letter = `안녕하세요 ${name} 입니다.`;

// backtick 장점(3) : 문자 중간 HTML 입력 가능.
var 변수명 = `변수`
var 문자 = `
<div>
	<div>
		${변수명}
	</div>
</div>
`
```

#### Template literals
```js
// 문자 중간중간에 있는 단어 순서를 바꾸거나 변수를 제거할 때 유용하다.
// 이로 문자 해체분석기능을 구현 가능하다.

var 변수 = '손흥민';

function 해체분석기(문자들, 변수들){
	console.log(문자들); // 출력 :  ['안녕하세요 ', ' 입니다', raw: Array(2)]
  	console.log(변수들); // 출력 : 손흥민
}
해체분석기`안녕하세요 ${변수} 입니다`; 
// 문자들 파라미터 : 파마미터들은 백틱 내의 순수 문자만 골라서 배열로 만들어준다.
// 변수들 파라미터 : 파마미터들은 백틱 내의 ${} 에 담기는 변수를 담아준다.
```
